```python
def query_at_position(pos: float) -> SegmentInfo:
```

### 2.2 车辆系统 (Vehicle System)

#### 2.2.1 职责

- 模拟列车动力学行为：牵引/惰行/制动三种工况
- 计算加速度、速度、位置随时间的变化
- 计算能耗和再生制动电量

#### 2.2.2 核心类设计

```python
class VehicleParams:
    empty_mass: float           # 空车质量（kg）
    passenger_capacity: int     # 满员载客数
    max_speed: float            # 最高速度（km/h）
    max_traction_force: float   # 最大牵引力（N）
    max_brake_force: float      # 最大制动力（N）
    davis_A: float              # Davis 公式 A 系数
    davis_B: float              # Davis 公式 B 系数
    davis_C_front_area: float   # 迎风面积（m²）
    davis_C_drag_coeff: float   # 空气阻力系数
    curve_resist_coeff: float   # 弯道阻力经验系数
    tunnel_resist_factor: float # 隧道阻力加成系数
    traction_curve: list[Point] # 牵引特性曲线折点列表
    regeneration_efficiency: float  # 再生制动效率（默认0.3）

class TrainState:
    position: float            # 当前公里标（m）
    speed: float               # 当前速度（km/h）
    acceleration: float        # 当前加速度（m/s²）
    mode: str                  # 工况：traction / coasting / braking
    mass: float                # 当前总质量（kg）
    passenger_count: int       # 当前载客数
    door_status: str           # 车门状态：open / closed / opening / closing
    fault_alarm: FaultAlarm    # 故障告警

class ControlCommands:
    traction_level: float      # 牵引级位 [0, 1]
    brake_level: float         # 制动级位 [0, 1]
    emergency_brake: bool      # 紧急制动

class TrainDynamics:
    """动力学解算引擎"""
    def step(self, state: TrainState, cmd: ControlCommands,
             track_params: SegmentInfo, dt: float) -> TrainState:
        # 1. 计算合力
        F = self._calc_net_force(state, cmd, track_params)
        # 2. 计算加速度：a = F / m
        a = F / state.mass
        # 3. 更新速度：v = v0 + a * dt
        # 4. 更新位置：x = x0 + v * dt
        # 5. 限速约束
        # 6. 能耗累计
        return new_state

    def _calc_net_force(self, state, cmd, track_params) -> float:
        # 牵引力
        F_trac = self._traction_force(state.speed, cmd.traction_level)
        # 制动力
        F_brake = self._brake_force(state.speed, cmd.brake_level)
        # 基本阻力（Davis 公式）
        R_base = self._davis_resistance(state.speed, state.mass)
        # 坡度附加阻力
        R_grad = self._gradient_resistance(state.mass, track_params.gradient)
        # 弯道附加阻力
        R_curv = self._curve_resistance(state.mass, track_params.curvature)
        # 隧道空气阻力
        R_tunnel = self._tunnel_resistance(R_base, track_params.is_tunnel)
        # 合力 = 牵引力 - 制动力 - 总阻力
        return F_trac - F_brake - (R_base + R_grad + R_curv + R_tunnel)
```

#### 2.2.3 核心算法

**牵引特性曲线**：分段线性插值，根据速度 `v` 和牵引级位 `λ` 计算牵引力。

```python
def _traction_force(self, speed_kmh: float, level: float) -> float:
    # 分段线性插值获取最大牵引力
    max_force = linear_interpolate(self.params.traction_curve, speed_kmh)
    # 乘以级位
    return max_force * level
```

**Davis 基本阻力公式**：

```
R_base = (A + B·v) × m·g + 0.5·ρ·Cd·Af·v²
```

其中：
- `A`、`B` 为 Davis 经验系数（已归一化至 ×m×g）
- `ρ` 为空气密度（1.225 kg/m³）
- `Cd` 为空气阻力系数
- `Af` 为迎风面积（m²）
- `v` 为速度（m/s）

**坡度附加阻力**：

```
R_grad = m · g · sin(θ) ≈ m · g · gradient / 1000
```

**弯道附加阻力**：

```
R_curv = k / curvature × m · g   （经验公式，k = 600）
```

#### 2.4.2 核心类设计

```python
class ThreeStageController:
    """三段式运行控制（迭代一 MVP）"""
    def compute_commands(self, state: TrainState, track_params: SegmentInfo,
                         timetable: Timetable) -> ControlCommands:
        # 三段式：牵引 → 惰行 → 制动
        # 制动点自动计算
        brake_point = self._calc_brake_point(state.speed, timetable.next_stop)
        if state.position >= brake_point:
            return ControlCommands(traction=0, brake=1.0)
        elif state.speed >= track_params.speed_limit * self.config.target_speed_ratio:
            return ControlCommands(traction=0, brake=0)  # 惰行
        else:
            return ControlCommands(traction=1.0, brake=0)  # 牵引

class ATPController:
    """ATP 安全防护"""
    safety_distance: float = 300         # 固定安全距离（m）
    overspeed_margin: float = 0.05       # 超速容忍 5%

    def check_safety(self, state: TrainState, track_params: SegmentInfo,
                     leading_train: TrainState) -> SafetyStatus:
        # 超速检测
        if state.speed > track_params.speed_limit * (1 + self.overspeed_margin):
            return SafetyStatus.EMERGENCY_BRAKE
        # 安全距离检测
        if leading_train and (leading_train.position - state.position) < self.safety_distance:
            return SafetyStatus.EMERGENCY_BRAKE
        return SafetyStatus.NORMAL

class ATOController:
    """ATO 自动驾驶（PID 控制）"""
    kp: float = 0.5
    ki: float = 0.01
    kd: float = 0.1

    def compute_commands(self, state: TrainState, target_speed: float) -> ControlCommands:
        error = target_speed - state.speed
        # PID 计算
        output = self.kp * error + self.ki * self.integral + self.kd * (error - self.prev_error)
        # 映射到牵引/制动级位
        if output > 0:
            return ControlCommands(traction=min(output, 1.0), brake=0)
        else:
            return ControlCommands(traction=0, brake=min(-output, 1.0))

class ATSController:
    """ATS 运行图调度（简化版）"""
    def adjust_dwell_time(self, train_state, timetable, actual_arrival_time) -> float:
        scheduled_departure = timetable.scheduled_departure
        delay = actual_arrival_time - scheduled_departure
        if delay > 0:
            # 晚点：缩短站停时间
            return max(timetable.min_dwell_time, timetable.nominal_dwell_time - delay)
        else:
            # 早点：恢复正常站停
            return timetable.nominal_dwell_time

class ManualDriveController:
    """手动驾驶控制器（迭代三）
    
    接收前端或实体司机台发送的手动控制指令，覆盖自动控制信号，
    实现司机手动操控列车牵引/制动、对标停车。
    """
    active: bool = False                # 手动驾驶模式是否激活
    manual_traction_level: float = 0.0  # 手动牵引级位 [0, 1]
    manual_brake_level: float = 0.0     # 手动制动级位 [0, 1]
    emergency_brake: bool = False       # 手动紧急制动

    def activate(self) -> None:
        """切入手动驾驶模式"""
        self.active = True

    def deactivate(self) -> None:
        """切回自动驾驶/三段式模式"""
        self.active = False
```

### 2.5 仿真编排器 (Orchestrator)

#### 2.5.1 职责

- 协调各子系统在每一仿真步中的执行顺序
- 管理仿真时钟、启停控制和速度倍率
- 通过 WebSocket 向前端推送状态

#### 2.5.2 核心类设计

```python
class Orchestrator:
    clock: SimulationClock
    track: TrackSystem
    vehicle: VehicleSystem
    power: PowerSystem
    signaling: SignalingSystem
    recorder: DataRecorder
    push_service: WebSocketPushService

    def __init__(self, config: SimulationConfig):
        self.clock = SimulationClock(config.time_step)
        self.track = TrackSystem(config.track_config)
        self.vehicle = VehicleSystem(config.vehicle_config)
        # ...

    def step(self) -> SimulationSnapshot:
        """执行一个仿真步"""
        # 1. 时钟推进
        self.clock.tick()

        # 2. 信号系统：计算控制指令
        sig_input = self._collect_signal_input()
        sig_output = self.signaling.step(sig_input)

        # 3. 轨道系统：查询线路参数
        track_input = self._collect_track_input(sig_output)
        track_output = self.track.step(track_input)

        # 4. 车辆系统：动力学解算
        veh_input = self._collect_vehicle_input(sig_output, track_output)
        veh_output = self.vehicle.step(veh_input)

        # 5. 供电系统：计算网压
        pwr_input = self._collect_power_input(veh_output)
        pwr_output = self.power.step(pwr_input)

        # 6. 数据记录
        snapshot = self._build_snapshot(sig_output, track_output, veh_output, pwr_output)
        self.recorder.record(snapshot)

        return snapshot

    def run(self):
        """主仿真循环"""
        while self.clock.elapsed < self.config.total_time and self.state == RunningState.RUNNING:
            snapshot = self.step()
            self.push_service.broadcast(snapshot)
            # 根据速度倍率等待
            sleep_time = self.clock.dt / self.config.speed_multiplier
            time.sleep(sleep_time)
```

| RUNNING | 正常运行 | PAUSED, STOPPED |
| PAUSED | 暂停 | RUNNING, STOPPED |

### 4.2 单列车站间运行流程

```
发车（站台） ──► 牵引加速 ──► 惰行 ──► 制动减速 ──► 对标停车（下一站）
    │              │              │            │              │
    │ 开门         │ 牵引力       │ 零牵引     │ 制动力       │ 速度=0，
    │ 等待站停时间  │ 满级位       │ 零制动     │ 按制动曲线   │ 位置在站台
    │ 关门         │ 速度上升     │ 速度缓慢   │ 速度下降     │ 中心±1m内
    │ 发车         │ 至目标速度   │ 下降       │ 至0          │ 开门
```

**制动点自动计算算法**：

```
制动距离 S_brake = v² / (2 × a_brake)    （a_brake 为制动减速度绝对值）
制动临界点 = 站台中心公里标 - S_brake - 安全余量
```

### 4.3 多车追踪流程（迭代三）

```
Train A ──► 位置 pos_A ──► 速度 v_A
Train B ──► 位置 pos_B ──► 速度 v_B

安全条件：pos_B - pos_A ≥ 安全距离（默认 500m）

ATP 检测：
  if pos_B - pos_A < 安全距离:
    Train B → 紧急制动
  elif pos_B - pos_A < 安全距离 + 制动距离:
    Train B → 限制牵引，准备制动
```

### 4.7 方案保存时的数据采集流程

```
保存时从 SimulationManager 提取：
  1. params = get_params()          → 车辆/信号/供电运行时参数
  2. config = get_config()          → simulation 参数（发车间隔、列车数等）
  3. 从最新 snapshot 提取：
     - totalConsumption             → tractionEnergy (kWh)
     - totalRegeneration            → regenEnergy (kWh)
  4. 从 recorder.summary() 提取：
     - total_time                   → totalTime (s)
     - avg_speed                    → avgSpeed (km/h)
     - max_speed                    → maxSpeed (km/h)
     - max_position                 → totalDistance (m)
  5. 从 voltageProfile 提取最小值   → minVoltage (V)
  6. 从 substations 提取最大值      → peakPower (kW)
```

### 5.3 WebSocket 连接管理

```typescript
// hooks/useWebSocket.ts
function useWebSocket(url: string) {
  const wsRef = useRef<WebSocket>(null);
  const reconnectTimer = useRef<number>(null);

  const connect = useCallback(() => {
    const ws = new WebSocket(url);
    ws.onopen = () => dispatch({ type: 'WS_CONNECTED' });
    ws.onmessage = (event) => {
      const snapshot = JSON.parse(event.data);
      dispatch({ type: 'RUNTIME_UPDATE', payload: snapshot });
    };
    ws.onclose = () => {
      dispatch({ type: 'WS_DISCONNECTED' });
      // 自动重连，间隔 2s
      reconnectTimer.current = setTimeout(connect, 2000);
    };
    wsRef.current = ws;
  }, [url]);

  useEffect(() => {
    connect();
    return () => {
      wsRef.current?.close();
      clearTimeout(reconnectTimer.current);
    };
  }, [connect]);

  return { send: wsRef.current?.send.bind(wsRef.current) };
}
```

```python
    @classmethod
```

| 决策 ID | 决策内容 | 理由 | 影响范围 |
|---------|----------|------|----------|
| ADR-01 | 一维公里标路径建模而非三维空间坐标 | 简化计算，满足轨道交通仿真需要 | 轨道系统 |

| ADR-04 | WebSocket 推全量快照而非增量差分 | 前端实现简单，避免状态同步 bug | 通信 |
| ADR-05 | 仿真引擎后端无 GUI，纯计算 + API | 可远程部署，便于 CI 测试 | 整体架构 |
| ADR-06 | 三段式运行模式作为 MVP 信号方案 | 无需复杂算法即可实现完整仿真闭环 | 信号系统 |
| ADR-07 | 固定网压作为迭代一供电方案 | 最小化供电系统复杂度，快速验证车辆动力学 | 供电系统 |
